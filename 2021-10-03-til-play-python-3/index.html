with open("kakao.csv", "r", encoding="utf-8") as f:
    lines = f.readlines()
    for line in lines:
        result = line.split(',"')
        if len(result) == 3:
            text += result[2].replace('ㅋ','').replace('ㅠ','').replace('채널추가하고','').replace('ㅜ','').replace('이모티콘','').replace('사진','').replace('"','')
